class OllamaLMSBridge {
  private $cache;

  public function processPrompt(string $query) {
    $hit = $this->cache->get(sha1($query));
    if ($hit) return $hit;

    $response = $this->ollama->generate($query);
    $this->cache->set(sha1($query), $response);
    return $response;
  }
}
